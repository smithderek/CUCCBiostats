Training setup

#########################
## Model Training/Tuning
#########################

numCV<-10 ## number of folds in cross validation

reSamp<-3 ## number of resamples

#create a list of seeds so that the results can be reproduced using parallel processing

seeds <- vector(mode = "list", length = (reSamp*numCV)+1)#length is = (n_repeats*nresampling)+1

set.seed(142)

for(i in 1:(reSamp*numCV)) seeds[[i]]<- sample.int(n=1000, size=10) #size must be >= to the number of tuning parameters

seeds[[(reSamp*numCV)+1]]<-sample.int(1000, 1)#for the last model when all the training data is fit to the selected tuning parameters




##############################################################################################################################
## Setup the training control parameters
##
## Resampling method: Cross Validation (method = "repeatedcv")
## Select number of folds in the CV (number = 10)
## Select class probabilities (classProbs = TRUE)
## Select the metric to summarize model fit and select the best training model (e.g. AUC) (summaryFunction = twoClassSummary)
## Allow for parallel processing (allowParallel=TRUE)
## Provide seeds so that results can be reproduced using parallel processing (seeds = seeds)
##############################################################################################################################

        ## Tuning parameters create the train control object for the CARET package
        fitControl <- trainControl(
          method = "repeatedcv",
          number = numCV,
          repeats= reSamp,
          classProbs = TRUE, ## needed to compute ROC metric
          summaryFunction = twoClassSummary, ## needed to compute ROC metric
          allowParallel=TRUE,
          seeds=seeds)


##############################################################################################################################

Other Measures of Classification Performance

performanceMetrics<-lapply(1:length(modPred),function(x){

## select the model results
 pred<-get(modPred[x])
  
## determine the Sen, Spe, PPV and NPV

## find the ROC cutpont based on Youden's Index
threshold<-ci.thresholds(rocPlots[[x]],thresholds = "best")
cutpoint<-as.numeric(rownames(threshold[[1]]))


## Extract the CI for the AUC
outCi<-round(ci(response=testing$SPM,predictor=pred$Yes),digits = 3)*100
outCi<-paste(outCi[2], " (",outCi[1], "-", outCi[3], ")",sep="")


## determine if the predicted class would be 1 or 0 using the cutpoint from the Youden's Index
classProb<-ifelse(pred$Yes>= cutpoint,1,0)

##  identify those with incorrect labels for the cases
FP<-ifelse(classProb==1 & testing$SPM!="Yes",1,0)

## determine the postive predictive value
PPV<-round(sum(testing$SPM=="Yes")/(sum(testing$SPM=="Yes")+sum(FP)),digits=3)*100


## identify those with incorrect labels for the controls
FN<-ifelse(classProb==0 & testing$SPM!="No",1,0)

## determine the negative predictive value
NPV<-round(sum(testing$SPM=="No")/(sum(testing$SPM=="No")+sum(FN)),digits=3)*100



## round the sensistivity ans specificity values
threshold$specificity<-round(threshold$specificity,digits=2)
threshold$sensitivity<-round(threshold$sensitivity,digits=2)

## output the threshold information
threshold<-cbind(paste(threshold$sensitivity[2]," (",threshold$sensitivity[1], "-", threshold$sensitivity[3], ")",sep=""),paste(threshold$specificity[2]," (",threshold$specificity[1], "-", threshold$specificity[3], ")",sep=""),PPV,NPV,round(cutpoint,digits=3))




performanceMetrics<-cbind("AUC"=outCi,threshold)

colnames(performanceMetrics)<- c("AUC % (95% CI)","Sensitivity % [Median (2.5 - 97.5%)]","Specificity % [Median (2.5 - 97.5%)]","PPV","NPV","ROC Cutpoint")
